String underscoreToCambel([String name = '']) {
  var response = '';
  for (var index = 0; index < name.length; index++) {
    var char = name[index];
    char = index == 0 && !isUpper(char) ? char.toUpperCase() : char;
    if (char == '_') {
      response = response +
          (index + 1 < name.length ? name[index + 1].toUpperCase() : '');
      index++;
    } else {
      response = response + char;
    }
  }
  return response;
}
